667  TrainingData<Scalar> &trainingData,
668  const MatrixConstPtr pQ)

671  // nlZ = (1/2) * (y-m)' * inv(K + D) * (y-m) + (1/2) * log |K + D|
672  // nlZ_j = (-1/2) * (y-m)' * inv(Kn) * D_j * inv(Kn) * (y-m) + (1/2) * tr[inv(Kn) * D_j]
673  // = (-1/2) * alpha' * D_j * alpha + (1/2) * tr[inv(Kn) * D_j]

676  // = (1/2) tr[Q * D_j]
677  return static_cast<Scalar>(0.5f) * pQ->cwiseProduct(Matrix(LikFunc<Scalar>::lik(logHyp, trainingData, 0)->asDiagonal())).sum();
